数据库重构

– 从旧推荐表迁移数据到新推荐表（处理可能的重复记录）
INSERT IGNORE INTO project_recommendations
(project_id, source_event_id, target_event_id, recommender_id,
recommendation_date, status)
SELECT
pro_id,
event_id,
COALESCE(rm_event_id, 0),
rm_user_id,
rm_time,
CASE
WHEN rm_status = 1 THEN ‘待审核’
WHEN rm_status = 2 THEN ‘已通过’
ELSE ‘待审核’
END
FROM pro_rm_event
WHERE pro_id > 0;

– 从竞赛推荐表迁移数据（处理可能的重复记录）
INSERT IGNORE INTO project_recommendations
(project_id, source_event_id, target_event_id, recommender_id,
recommendation_date, status)
SELECT
pro_id,
pro_event_id,
recommended_event_id,
0, – 无法确定推荐人，置为0
created_at,
CASE
WHEN recommendation_status = ‘成功推荐’ THEN ‘已通过’
ELSE ‘待审核’
END
FROM competition_recommendation
WHERE pro_id > 0;
